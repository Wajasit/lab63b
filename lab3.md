## วิธีการทำการทดลอง 
1. ทำการต่อ อแดปเตอร์เข้ากับซีเรียลก่อน แล้วจึงต่อไมโครคอนโทรลเลอร์ ESP-01 บนตัวอแดปเตอร์
2. อัพโหลดโปรแกรมลงไมโครคอนโทรลเลอร์ โดยใช้คำสั่ง pio run -t upload หลังจากนั้นก็ทำการกดปุ่มสองปุ่มที่ตัวซีเรียล เพื่อทำการรับการอัพโหลดโปรแกรม

		#include <Arduino.h>
		#include <ESP8266WiFi.h>

		int cnt = 0;

		void setup()
		{
			Serial.begin(115200);
			pinMode(0, OUTPUT);
			Serial.println("\n\n\n");
		}

		void loop()
		{
			cnt++;
			if(cnt % 2) {
				Serial.println("========== ON ===========");
				digitalWrite(0, HIGH);
			} else {
				Serial.println("========== OFF ===========");
				digitalWrite(0, LOW);
			}
			delay(500);
		}

## อภิปรายผลการทดลอง 
จากการทดลองจะเห็นว่าที่พอตศูนย์ไฟจะเปิด-ปิด ตามโปรแกรมที่เขียนไว้ว่าจำนวนนับคี่ไฟจะติดแต่ถ้าเป็นคู่ไฟจะปิด และเมื่อกดรีเซ็ตก็จะเริ่มนับใหม่
## คำถามหลังการทดลอง 
* Q:delay 500 คืออะไร
* A:จะทำให้สัญญาณว้นระยะเวลาไป500ms หรือ ทุกๆครึ่งวินาที จะเกิดการวนลูปนั่นเอง
